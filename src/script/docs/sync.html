    initEvents: ->
      @onStorage = (evt) =>

    set: (key, value) ->
      if key not in @validKeys

      state[key] = value

    get: (key) ->
      if key not in @validKeys
        throw new Error "Can't get #{key} from storage; not a valid key"
      state = @getState()
      return switch key
        when 'file' then state.file
        when 'slide' then Number state.slide
        when 'hidden' then /^true$/i.test state.hidden
